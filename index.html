Cisco IOS – VLAN/SVI, DHCP, ACL (exemple)
! VLANs & SVI (switch L3)
conf t
vlan 10
 name USERS
vlan 20
 name PRINTERS
vlan 30
 name VIDEO
vlan 99
 name MGMT
!
interface Vlan10
 ip address 192.168.10.1 255.255.255.0
 no shut
interface Vlan20
 ip address 192.168.20.1 255.255.255.0
 no shut
interface Vlan30
 ip address 192.168.30.1 255.255.255.0
 no shut
interface Vlan99
 ip address 192.168.99.1 255.255.255.0
 no shut
!
! Trunk vers switch d'accès / AP
interface GigabitEthernet1/0/48
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30,99
!
! DHCP locaux (optionnel)
ip dhcp excluded-address 192.168.10.1 192.168.10.49
ip dhcp pool VLAN10
 network 192.168.10.0 255.255.255.0
 default-router 192.168.10.1
 dns-server 192.168.10.10
!
! ACL: autoriser Users -> Printers uniquement TCP/9100,515,631
ip access-list extended USERS-TO-PRN
 permit tcp 192.168.10.0 0.0.0.255 192.168.20.0 0.0.0.255 eq 9100
 permit tcp 192.168.10.0 0.0.0.255 192.168.20.0 0.0.0.255 eq 515
 permit tcp 192.168.10.0 0.0.0.255 192.168.20.0 0.0.0.255 eq 631
 deny   ip  192.168.10.0 0.0.0.255 192.168.20.0 0.0.0.255
 permit ip  any any
!
interface Vlan20
 ip access-group USERS-TO-PRN in
end
wr mem
PowerShell – imprimantes & tests
# Ajouter une imprimante TCP/IP + pilote
$ip = "192.168.20.25"
$port = "IP_$ip"
Add-PrinterPort -Name $port -PrinterHostAddress $ip
Add-PrinterDriver -Name "HP Universal Printing PCL 6" -ErrorAction SilentlyContinue
Add-Printer -Name "HP-Repro-01" -DriverName "HP Universal Printing PCL 6" -PortName $port

# Test de connectivité rapide
Test-Connection -ComputerName $ip -Count 3

# Partage d'imprimante depuis un hôte d'impression
Set-Printer -Name "HP-Repro-01" -Shared $true -ShareName "HP-Repro-01"
Active Directory – création utilisateur & groupage
Import-Module ActiveDirectory

# Créer un utilisateur standard dans une OU
New-ADUser -Name "Jean Dupont" -GivenName Jean -Surname Dupont `
  -SamAccountName jdupont -UserPrincipalName j.dupont@entreprise.local `
  -Path "OU=Users,OU=Site-A,DC=entreprise,DC=local" -Enabled $true `
  -ChangePasswordAtLogon $true

# Ajouter au groupe d'impression ou service
Add-ADGroupMember -Identity "GRP-Impression" -Members jdupont

  • Imprimantes : vérifier file d’attente locale, port TCP/IP, pilote, puis partage.
